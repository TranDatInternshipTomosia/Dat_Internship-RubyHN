----- Các phép toán số học trong Ruby
	- Các toán tử số học Ruby phổ biến nhất là cộng (+), trừ (-), chia lấy phần số (/), nhân (*), lũy thừa (**) và chia lấy phần dư (%).
	vidu: print 1+3
		# Addition: output 4

		print 1-2
		# Subtraction: output -1

		print 9/3
		# Division: output 3

		print 2*3
		# Multiplication: output 6

		print 2**3
		# Exponentiation: output 8

		print 16%9
		# Modulo: output 7
